# Frontend Environment Variables (.env.development.local or .env.production)
# OPTIONAL - The frontend automatically detects the correct API URL

# For local development, NO .env file is needed!
# The system auto-detects:
#   - localhost → https://localhost:5001
#   - network IP → https://[SAME_IP]:5001

# Only create .env.development.local if you need to override auto-detection:
# VITE_API_BASE_URL=https://localhost:5001

# For shared environments with custom URLs:
# VITE_API_BASE_URL=https://192.168.1.100:5001
# VITE_API_BASE_URL=https://your-domain.com:5001
