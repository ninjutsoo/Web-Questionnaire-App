rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Allow public lookups on your "Users" collection
    // (change "Users" → exactly the same capitalization you use in code)
    match /Users/{userId} {
      // Allow both single‐document `get` and collection `list` queries
      allow get, list: if true;
    }
 
    // All other reads/writes still require an authenticated user
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
